# OLT Raisecom ISCOM 5508 OLT-gp4a (ROAP). После enable запрашивает пароль — указать в secret.
# {port} = ONU ID (например 3/1/5), {vlan} = VLAN клиента.
# В команде gpon-olt 3/1 при необходимости замените 3/1 на свой слот/порт OLT.
device_type=raisecom_roap
username=admin
password=admin
secret=admin
---
enable
sh int gpon-onu {port} online-information
sh gpon-onu {port} tr
sh int gpon-olt 3/1 transceiver rx-onu-power | include {port}
sh gpon-onu {port} uni ethernet information
sh ver
sh mac-address-table l2 vlan {vlan}
sh mac-address-table l2 interface gpon-onu {port}
